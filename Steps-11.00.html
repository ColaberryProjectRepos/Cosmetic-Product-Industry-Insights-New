DAX Formula for each Measure

1. Arrow_Chem_YoY %
Arrow_Chem_YoY % = 
VAR PreviousYearCount = 
    CALCULATE(
        DISTINCTCOUNT('cscpopendata'[ChemicalId]),
        DATEADD('Calendar'[Date], -1, YEAR)
    )
VAR YoYPercentage = 
    DIVIDE(
        DISTINCTCOUNT('cscpopendata'[ChemicalId]) - PreviousYearCount,
        PreviousYearCount
    )
RETURN
    IF(
        YoYPercentage <= 0,
        UNICHAR(9660), -- Down arrow (negative or zero)
        UNICHAR(9650)  -- Up arrow (positive)
    )

VAR PreviousYearCount = 
    CALCULATE(
        'Main Measures'[DPC],
        DATEADD('Calendar'[Date], -1, YEAR)
    )
VAR YoYPercentage = 
    DIVIDE(
        IF(ISBLANK('Main Measures'[DPC]), 0, 'Main Measures'[DPC]) - PreviousYearCount,
        PreviousYearCount
    )
RETURN
    IF(
        [DP_YoY %] <= 0 || ISBLANK('Main Measures'[DPC]),

        UNICHAR(9650)  -- Up arrow (positive)
    )

5. Date_Slicer_Change
Date_Slicer_Change = 
IF(
    ISFILTERED('cscpopendata'[InitialReportedYear]),
    IF(
        SELECTEDVALUE('cscpopendata'[InitialReportedYear], 0) <> 0,
        1,
        0
    ),
    BLANK()
)
6. Dynamic Chart Title Selected Value
Dynamic Chart Title Selected Value = 
VAR Yr = 
    IF(
        ISFILTERED('Calendar'[Date].[Year]),
        "of " & SELECTEDVALUE('Calendar'[Date].[Year]) & " by Month"
    )

RETURN     "Product Count " & Yr
7. Chemical Count
Chemical Count = COUNTROWS(DISTINCT('cscpopendata'[ChemicalId]))
8. Company Count
Company Count = COUNTROWS(DISTINCT('cscpopendata'[CompanyId]))
9. DPC
DPC = 
CALCULATE (
    COUNTROWS ( DISTINCT ( 'cscpopendata'[ProductName] ) ),
    NOT ISBLANK ( 'cscpopendata'[DiscontinuedDate] ),
    NOT ISBLANK ( 'cscpopendata'[InitialDateReported] ),
    'cscpopendata'[DiscontinuedDate] > DATE(2008, 1, 1)
)
10. Discontinued_Product_Percentage
Discontinued_Product_Percentage = 

VAR Percentage =
    DIVIDE([DPC], [Product Count]) 
RETURN
    Percentage
11. Product Count
Product Count = CALCULATE(
        COUNTROWS(DISTINCT(cscpopendata[ProductName])),
        'cscpopendata'[InitialDateReported] <> BLANK()
    )
